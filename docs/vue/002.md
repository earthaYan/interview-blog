# vue2-组件使用

## 父子组件通信

通过 props 和 this.$emit 方法实现父子组件通信
::: code-group

```vue [父组件]
<template>
  <dv>
    <Input @add="addHandler" />
    <List :list="list" @delete="deleteHandler" />
  </dv>
</template>
<script>
import Input from "./SearchInput.vue";
import List from "./ResultList.vue";
export default {
  name: "ComponentDemo",
  components: { Input, List },
  data() {
    return {
      list: [{ id: 1, title: "todo1" }],
    };
  },
  methods: {
    addHandler(title) {
      this.list.push({ id: `id-${Date.now()}`, title });
    },
    deleteHandler(id) {
      this.list = this.list.filter((item) => item.id !== id);
    },
  },
};
</script>
```

```vue [Input组件]
<template>
  <div>
    <input v-model="content" />
    <button @click="addTitle">添加</button>
  </div>
</template>
<script>
export default {
  data() {
    return {
      content: "",
    };
  },
  methods: {
    addTitle() {
      this.$emit("add", this.content);
    },
  },
};
</script>
```

```vue [List组件]
<template>
  <ul>
    <li v-for="item in list" :key="item.id">
      {{ item.title }}
      <button @click="deleteItem(item.id)">删除</button>
    </li>
  </ul>
</template>
<script>
export default {
  props: {
    list: {
      type: Array,
      default: () => [],
    },
  },
  methods: {
    deleteItem(id) {
      this.$emit("delete", id);
    },
  },
};
</script>
```

## 兄弟组件通信

Vue 本身具有自定义事件的能力，且 vue 实例支持事件绑定和监听和解绑

::: code-group

```js [event.js]
import Vue from "vue";
export default new Vue({});
```

```vue [Input组件]
<template>
  <div>
    <input type="text" v-model="title" />
    <button @click="addTitle">add</button>
  </div>
</template>

<script>
import event from "./event";

export default {
  data() {
    return {
      title: "",
    };
  },
  methods: {
    addTitle() {
      // 调用自定义事件
      event.$emit("onAddTitle", this.title);
      this.title = "";
    },
  },
};
</script>
```

```vue [List组件]
<template>
  <ul>
    <li v-for="item in list" :key="item.id">
      {{ item.title }}
      <button @click="deleteItem(item.id)">删除</button>
    </li>
  </ul>
</template>
<script>
import event from "./event";
export default {
  props: {
    list: {
      type: Array,
      default: () => [],
    },
  },
  methods: {
    deleteItem(id) {
      this.$emit("delete", id);
    },
    addHandler(title) {
      console.log("on add title", title);
    },
  },
  mounted() {
    // 绑定事件
    event.$on("onAddTitle", this.addHandler);
  },
  beforeDestroy() {
    //及时销毁，避免造成内存泄漏
    event.$off("onAddTitle", this.addHandler);
  },
};
</script>
```

## 组件生命周期

### 单个组件

- 挂载：beforeCreate,created,beforeMount.mounted
- 更新：beforeUpdate,updated
- 销毁：beforeDestroy,destroyed

#### mounted 和 created 区别

- created：数据已完成初始化，但 DOM 还没生成。
- mounted：DOM 已经渲染到页面，可以操作真实 DOM。
